<script>
    
    //Making an object 
    let obj = {
        id:1,
        Name: "Farhan"
    };

    // add new property
    obj.F_Name = "Qamar Uddin";

    obj.id=2; // update property
    
    // delete property
    delete obj.Name;

    // object with array
    let arr = [
        {id:1, Name: "Farhan", F_Name: "Qamar"},
        {id:2, Name: "Hassan", F_Name: "Zafar"},
        {id:3, Name: "Mazhar", F_Name: "Yaseen"},
        {id:4, Name: "Mohsin", F_Name: "Faheem"}
    ];
    
    // Pushing a new object in array

    let new1 = {
        id: 1,
        Name: "Ali",
        F_name: "Liaqut"
    };
    arr.push(new1);
    
    // Updating id with the value of (loop)
    
    for (let i=0; i<arr.length; i++){
        arr[i].id=i;
    }

    // Merging two objects using spread
    let Merge1 = { city: "Karachi", country: "Pakistan" };
    let Merge2 = { occupation: "Developer", age: 25 };
    
    let mergedObj = { ...Merge1, ...Merge2 };
    
    console.log("Original Object :", obj);
    console.log("Final Array after push and ID update:", arr);
    console.log("Newly Merged Object (using spread):", mergedObj);
</script>
